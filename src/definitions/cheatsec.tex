\begin{definition}[Cheating Security]
\label{def:cheatsec}
  We say that a protocol is secure against cheating if $\forall \: \mathrm{PPT} \:
  \mathcal{E}, \Pr\left[\mathcal{E} \: \mathrm{receiving} \: \left(\mathtt{cheated, \_,
  \_}\right)\right]$ is negligible.
\end{definition}
